CREATE TABLE Usuario
(id_user number GENERATED BY DEFAULT ON NULL AS IDENTITY,
username varchar(20),
password varchar(250),
profesor char(1));


CREATE TABLE Alumno
(id_alumno int GENERATED BY DEFAULT ON NULL AS IDENTITY,
id_user number,
nombre varchar(50),
apellido varchar(25),
expediente varchar(8),
carrera varchar(50),
clase number
foto blob);


CREATE TABLE Profesor
(id_profesor int GENERATED BY DEFAULT ON NULL AS IDENTITY,
id_user number,
nombre varchar(25),
apellido varchar(25),
expediente varchar(8),
carrera varchar(50),
clase number
foto blob);


CREATE TABLE Notas
(id_notas number GENERATED BY DEFAULT ON NULL AS IDENTITY,
id_alumno number,
id_asignatura number,
nota number,
asignatura varchar(50),
prueba varchar(50));

CREATE TABLE Registro
(id_registro number GENERATED BY DEFAULT ON NULL AS IDENTITY,
aula number,
temperatura number,
ruido number,
humedad number,
hora timestamp);

CREATE TABLE Carrera
(id_carrera number GENERATED BY DEFAULT ON NULL AS IDENTITY,
carrera varchar(50),
precio number);

CREATE TABLE Asignaturas
(id_asignaturas number GENERATED BY DEFAULT ON NULL AS IDENTITY,
id_carrera number,
codigoAsignatura number,
asignatura varchar(50),
creditos number,
tipo varchar(50),
curso number);
