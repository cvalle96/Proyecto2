CREATE TABLE Usuario
(id_user number GENERATED BY DEFAULT ON NULL AS IDENTITY,
username varchar(8),
password varchar(250),
profesor char(1));


CREATE TABLE Alumno
(id_alumno int GENERATED BY DEFAULT ON NULL AS IDENTITY,
id_user number,
nombre varchar(50),
apellido varchar(25),
expediente varchar(8),
carrera varchar(50),
clase number);


CREATE TABLE Profesor
(id_profesor int GENERATED BY DEFAULT ON NULL AS IDENTITY,
id_user number,
nombre varchar(25),
apellido varchar(25),
expediente varchar(8),
carrera varchar(50),
clase number);


CREATE TABLE Notas
(id_notas number GENERATED BY DEFAULT ON NULL AS IDENTITY,
id_alumno number,
nota number,
asignatura varchar(50),
prueba varchar(50));

CREATE TABLE Registro
(id_registro number GENERATED BY DEFAULT ON NULL AS IDENTITY,
clase number,
temperatura number,
ruido number,
humedad number,
hora Date);
