CREATE TABLE Usuario
(id_user number GENERATED BY DEFAULT ON NULL AS IDENTITY,
expediente varchar(50),
contraseña varchar(50),
nombre varchar(50),
apellido varchar(25),
telefono varchar(25),
profesor char(1));


CREATE TABLE Alumno
(id_alumno int GENERATED BY DEFAULT ON NULL AS IDENTITY,
id_user number,
expediente number(8),
contraseña varchar(50),
nombre varchar(50),
apellido varchar(25),
telefono varchar(25));


CREATE TABLE Profesor
(id_profesor int GENERATED BY DEFAULT ON NULL AS IDENTITY,
id_user number,
grupos varchar(4),
asignatura varchar(50));


CREATE TABLE Notas
(id_notas number GENERATED BY DEFAULT ON NULL AS IDENTITY,
id_alumno number,
nota number,
asignatura varchar(50),
prueba varchar(50));


CREATE TABLE Aula
(id_aula number GENERATED BY DEFAULT ON NULL AS IDENTITY,
id_grupo number,
ordenadores char(1));


CREATE TABLE Registro
(id_registro number GENERATED BY DEFAULT ON NULL AS IDENTITY,
id_aula number,
temperatura number,
ruido number,
humedad number,
hora Date);


CREATE TABLE Grupo
(id_grupo number GENERATED BY DEFAULT ON NULL AS IDENTITY,
id_alumno number,
carrera varchar(50));
